import org.apache.activemq.artemis.jms.client.ActiveMQJMSClient;
import javax.jms.TopicConnectionFactory;

public TopicConnectionFactory getTopicConnectionFactory() throws JMSException {
    TransportConfiguration transportConfiguration = new TransportConfiguration(NettyConnectorFactory.class.getName());
    TopicConnectionFactory connectionFactory = ActiveMQJMSClient.createConnectionFactoryWithoutHA(JMSFactoryType.CF, transportConfiguration);
    return connectionFactory;
}
